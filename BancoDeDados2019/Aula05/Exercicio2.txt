create or replace procedure PEDIDOEVALORTOTAL(P_COD IN PEDIDO.CODPEDIDO%TYPE, P_QUANTIDADE OUT ITEMPEDIDO.QUANTIDADE%TYPE, P_VALORTOTAL OUT NUMBER ) IS
BEGIN
  SELECT (VALORUNITARIO * QUANTIDADE), QUANTIDADE INTO P_VALORTOTAL, P_QUANTIDADE FROM ITEMPEDIDO WHERE CODPEDIDO = P_COD;
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    P_QUANTIDADE := 0;
    P_VALORTOTAL :=0;
END;

SELECT PEDIDOEAVALORTOTAL(1, a, B) FROM DUAL;
